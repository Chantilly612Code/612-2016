58  /**
59  * Enumerated types for Motion Control Set Values.
60  * When in Motion Profile control mode, these constants are paseed
61  * into set() to manipulate the motion profile executer.
62  * When changing modes, be sure to read the value back using getMotionProfileStatus()
63  * to ensure changes in output take effect before performing buffering actions.
64  * Disable will signal Talon to put motor output into neutral drive.
65  * Talon will stop processing motion profile points. This means the buffer is
66  * effectively disconnected from the executer, allowing the robot to gracefully
67  * clear and push new traj points. isUnderrun will get cleared.
68  * The active trajectory is also cleared.
69  * Enable will signal Talon to pop a trajectory point from it's buffer and process it.
70  * If the active trajectory is empty, Talon will shift in the next point.
71  * If the active traj is empty, and so is the buffer, the motor drive is neutral and
72  * isUnderrun is set. When active traj times out, and buffer has at least one point,
73  * Talon shifts in next one, and isUnderrun is cleared. When active traj times out,
74  * and buffer is empty, Talon keeps processing active traj and sets IsUnderrun.
75  * Hold will signal Talon keep processing the active trajectory indefinitely.
76  * If the active traj is cleared, Talon will neutral motor drive. Otherwise
77  * Talon will keep processing the active traj but it will not shift in
78  * points from the buffer. This means the buffer is effectively disconnected
79  * from the executer, allowing the robot to gracefully clear and push
80  * new traj points.
81  * isUnderrun is set if active traj is empty, otherwise it is cleared.
82  * isLast signal is also cleared.
83  *
84  * Typical workflow:
85  * set(Disable),
86  * Confirm Disable takes effect,
87  * clear buffer and push buffer points,
88  * set(Enable) when enough points have been pushed to ensure no underruns,
89  * wait for MP to finish or decide abort,
90  * If MP finished gracefully set(Hold) to hold position servo and disconnect buffer,
91  * If MP is being aborted set(Disable) to neutral the motor and disconnect buffer,
92  * Confirm mode takes effect,
93  * clear buffer and push buffer points, and rinse-repeat.
94  */
95  enum SetValueMotionProfile {
96  SetValueMotionProfileDisable = 0,
97  SetValueMotionProfileEnable = 1,
98  SetValueMotionProfileHold = 2,
99  };
100  /**
101  * Motion Profile Trajectory Point
102  * This is simply a data transer object.
103  */
104  struct TrajectoryPoint {
105  double position; //!< The position to servo to.
106  double velocity; //!< The velocity to feed-forward.
107  /**
108  * Time in milliseconds to process this point.
109  * Value should be between 1ms and 255ms. If value is zero
110  * then Talon will default to 1ms. If value exceeds 255ms API will cap it.
111  */
112  unsigned int timeDurMs;
113  /**
114  * Which slot to get PIDF gains.
115  * PID is used for position servo.
116  * F is used as the Kv constant for velocity feed-forward.
117  * Typically this is hardcoded to the a particular slot, but you are free
118  * gain schedule if need be.
119  */
120  unsigned int profileSlotSelect;
121  /**
122  * Set to true to only perform the velocity feed-forward and not perform
123  * position servo. This is useful when learning how the position servo
124  * changes the motor response. The same could be accomplish by clearing the
125  * PID gains, however this is synchronous the streaming, and doesn't require restoing
126  * gains when finished.
127  *
128  * Additionaly setting this basically gives you direct control of the motor output
129  * since motor output = targetVelocity X Kv, where Kv is our Fgain.
130  * This means you can also scheduling straight-throttle curves without relying on
131  * a sensor.
132  */
133  bool velocityOnly;
134  /**
135  * Set to true to signal Talon that this is the final point, so do not
136  * attempt to pop another trajectory point from out of the Talon buffer.
137  * Instead continue processing this way point. Typically the velocity
138  * member variable should be zero so that the motor doesn't spin indefinitely.
139  */
140  bool isLastPoint;
141  /**
142  * Set to true to signal Talon to zero the selected sensor.
143  * When generating MPs, one simple method is to make the first target position zero,
144  * and the final target position the target distance from the current position.
145  * Then when you fire the MP, the current position gets set to zero.
146  * If this is the intent, you can set zeroPos on the first trajectory point.
147  *
148  * Otherwise you can leave this false for all points, and offset the positions
149  * of all trajectory points so they are correct.
150  */
151  bool zeroPos;
152  };
