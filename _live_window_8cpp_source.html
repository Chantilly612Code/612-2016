33 /**
34  * Change the enabled status of LiveWindow
35  * If it changes to enabled, start livewindow running otherwise stop it
36  */
37 void LiveWindow::SetEnabled(bool enabled) {
38  if (m_enabled == enabled) return;
39  if (enabled) {
40  if (m_firstTime) {
41  InitializeLiveWindowComponents();
42  m_firstTime = false;
43  }
44  m_scheduler->SetEnabled(false);
45  m_scheduler->RemoveAll();
46  for (auto& elem : m_components) {
47  elem.first->StartLiveWindowMode();
48  }
49  } else {
50  for (auto& elem : m_components) {
51  elem.first->StopLiveWindowMode();
52  }
53  m_scheduler->SetEnabled(true);
54  }
55  m_enabled = enabled;
56  m_statusTable->PutBoolean("LW Enabled", m_enabled);
57 }

197 /**
198  * Initialize all the LiveWindow elements the first time we enter LiveWindow
199  * mode. By holding off creating the NetworkTable entries, it allows them to be
200  * redefined before the first time in LiveWindow mode. This allows default
201  * sensor and actuator values to be created that are replaced with the custom
202  * names from users calling addActuator and addSensor.
203  */
204 void LiveWindow::InitializeLiveWindowComponents() {
205  for (auto& elem : m_components) {
206  std::shared_ptr<LiveWindowSendable> component = elem.first;
207  LiveWindowComponent c = elem.second;
208  std::string subsystem = c.subsystem;
209  std::string name = c.name;
210  m_liveWindowTable->GetSubTable(subsystem)
211  ->PutString("~TYPE~", "LW Subsystem");
212  std::shared_ptr<ITable> table(
213  m_liveWindowTable->GetSubTable(subsystem)->GetSubTable(name));
214  table->PutString("~TYPE~", component->GetSmartDashboardType());
215  table->PutString("Name", name);
216  table->PutString("Subsystem", subsystem);
217  component->InitTable(table);
218  if (c.isSensor) {
219  m_sensors.push_back(component);
220  }
221  }
222 }
