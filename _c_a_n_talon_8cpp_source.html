98 float CANTalon::Get() const {
99  int value;
100  switch (m_controlMode) {
101  case kVoltage:
102  return GetOutputVoltage();
103  case kCurrent:
104  return GetOutputCurrent();
105  case kSpeed:
106  m_impl->GetSensorVelocity(value);
107  return ScaleNativeUnitsToRpm(m_feedbackDevice, value);
108  case kPosition:
109  m_impl->GetSensorPosition(value);
110  return ScaleNativeUnitsToRotations(m_feedbackDevice, value);
111  case kPercentVbus:
112  case kFollower:
113  default:
114  m_impl->GetAppliedThrottle(value);
115  return (float)value / 1023.0;
116  }

119 /**
120  * Sets the appropriate output on the talon, depending on the mode.
121  *
122  * In PercentVbus, the output is between -1.0 and 1.0, with 0.0 as stopped.
123  * In Voltage mode, output value is in volts.
124  * In Current mode, output value is in amperes.
125  * In Speed mode, output value is in position change / 10ms.
126  * In Position mode, output value is in encoder ticks or an analog value,
127  * depending on the sensor.
128  * In Follower mode, the output value is the integer device ID of the talon to
129  * duplicate.
130  *
131  * @param outputValue The setpoint value, as described above.
132  * @see SelectProfileSlot to choose between the two sets of gains.
133  */
134 void CANTalon::Set(float value, uint8_t syncGroup) {
135  /* feed safety helper since caller just updated our output */
136  m_safetyHelper->Feed();
137 
138  if (m_stopped) {
139  EnableControl();
140  m_stopped = false;
141  }
142 
143  if (m_controlEnabled) {
144  m_setPoint = value; /* cache set point for GetSetpoint() */
145  CTR_Code status = CTR_OKAY;
146  switch (m_controlMode) {
147  case CANSpeedController::kPercentVbus: {
148  m_impl->Set(m_isInverted ? -value : value);
149  status = CTR_OKAY;
150  } break;
151  case CANSpeedController::kFollower: {
152  status = m_impl->SetDemand((int)value);
153  } break;
154  case CANSpeedController::kVoltage: {
155  // Voltage is an 8.8 fixed point number.
156  int volts = int((m_isInverted ? -value : value) * 256);
157  status = m_impl->SetDemand(volts);
158  } break;
159  case CANSpeedController::kSpeed:
160  /* if the caller has provided scaling info, apply it */
161  status = m_impl->SetDemand(ScaleVelocityToNativeUnits(m_feedbackDevice, m_isInverted ? -value : value));
162  break;
163  case CANSpeedController::kPosition:
164  status = m_impl->SetDemand(ScaleRotationsToNativeUnits(m_feedbackDevice, value));
165  break;
166  case CANSpeedController::kCurrent: {
167  double milliamperes = (m_isInverted ? -value : value) * 1000.0; /* mA*/
168  status = m_impl->SetDemand(milliamperes);
169  } break;
170  case CANSpeedController::kMotionProfile: {
171  status = m_impl->SetDemand((int)value);
172  } break;
173  default:
174  wpi_setWPIErrorWithContext(
175  IncompatibleMode,
176  "The CAN Talon does not support this control mode.");
177  break;
178  }
179  if (status != CTR_OKAY) {
180  wpi_setErrorWithContext(status, getHALErrorMessage(status));
181  }
182 
183  status = m_impl->SetModeSelect(m_sendMode);
184 
185  if (status != CTR_OKAY) {
186  wpi_setErrorWithContext(status, getHALErrorMessage(status));
187  }
188  }
189 }

1558 /**
1559  * @param devToLookup FeedbackDevice to lookup the scalar for. Because Talon
1560  * allows multiple sensors to be attached simultaneously, caller must
1561  * specify which sensor to lookup.
1562  * @return The number of native Talon units per rotation of the selected sensor.
1563  * Zero if the necessary sensor information is not available.
1564  * @see ConfigEncoderCodesPerRev
1565  * @see ConfigPotentiometerTurns
1566  */
1567 double CANTalon::GetNativeUnitsPerRotationScalar(FeedbackDevice devToLookup)const
1568 {
1569  bool scalingAvail = false;
1570  CTR_Code status = CTR_OKAY;
1571  double retval = 0;
1572  switch (devToLookup) {
1573  case QuadEncoder:
1574  { /* When caller wants to lookup Quadrature, the QEI may be in 1x if the selected feedback is edge counter.
1575  * Additionally if the quadrature source is the CTRE Mag encoder, then the CPR is known.
1576  * This is nice in that the calling app does not require knowing the CPR at all.
1577  * So do both checks here.
1578  */
1579  int32_t qeiPulsePerCount = 4; /* default to 4x */
1580  switch (m_feedbackDevice) {
1581  case CtreMagEncoder_Relative:
1582  case CtreMagEncoder_Absolute:
1583  /* we assume the quadrature signal comes from the MagEnc,
1584  of which we know the CPR already */
1585  retval = kNativePwdUnitsPerRotation;
1586  scalingAvail = true;
1587  break;
1588  case EncRising: /* Talon's QEI is setup for 1x, so perform 1x math */
1589  case EncFalling:
1590  qeiPulsePerCount = 1;
1591  break;
1592  case QuadEncoder: /* Talon's QEI is 4x */
1593  default: /* pulse width and everything else, assume its regular quad use. */
1594  break;
1595  }
1596  if (scalingAvail) {
1597  /* already deduced the scalar above, we're done. */
1598  } else {
1599  /* we couldn't deduce the scalar just based on the selection */
1600  if (0 == m_codesPerRev) {
1601  /* caller has never set the CPR. Most likely caller
1602  is just using engineering units so fall to the
1603  bottom of this func.*/
1604  } else {
1605  /* Talon expects PPR units */
1606  retval = qeiPulsePerCount * m_codesPerRev;
1607  scalingAvail = true;
1608  }
1609  }
1610  } break;
1611  case EncRising:
1612  case EncFalling:
1613  if (0 == m_codesPerRev) {
1614  /* caller has never set the CPR. Most likely caller
1615  is just using engineering units so fall to the
1616  bottom of this func.*/
1617  } else {
1618  /* Talon expects PPR units */
1619  retval = 1 * m_codesPerRev;
1620  scalingAvail = true;
1621  }
1622  break;
1623  case AnalogPot:
1624  case AnalogEncoder:
1625  if (0 == m_numPotTurns) {
1626  /* caller has never set the CPR. Most likely caller
1627  is just using engineering units so fall to the
1628  bottom of this func.*/
1629  } else {
1630  retval = (double)kNativeAdcUnitsPerRotation / m_numPotTurns;
1631  scalingAvail = true;
1632  }
1633  break;
1634  case CtreMagEncoder_Relative:
1635  case CtreMagEncoder_Absolute:
1636  case PulseWidth:
1637  retval = kNativePwdUnitsPerRotation;
1638  scalingAvail = true;
1639  break;
1640  }
1641  /* handle any detected errors */
1642  if (status != CTR_OKAY) {
1643  wpi_setErrorWithContext(status, getHALErrorMessage(status));
1644  }
1645  /* if scaling information is not possible, signal caller
1646  by returning zero */
1647  if (false == scalingAvail)
1648  retval = 0;
1649  return retval;
1650 }
