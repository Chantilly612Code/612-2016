51 /**
52  * Get the acceleration of one axis in Gs.
53  *
54  * @param axis The axis to read from.
55  * @return Acceleration of the ADXL345 in Gs.
56  */
57 double ADXL345_I2C::GetAcceleration(ADXL345_I2C::Axes axis) {
58  int16_t rawAccel = 0;
59  Read(kDataRegister + (uint8_t)axis, sizeof(rawAccel), (uint8_t *)&rawAccel);
60  return rawAccel * kGsPerLSB;
61 }

63 /**
64  * Get the acceleration of all axes in Gs.
65  *
66  * @return An object containing the acceleration measured on each axis of the
67  * ADXL345 in Gs.
68  */
69 ADXL345_I2C::AllAxes ADXL345_I2C::GetAccelerations() {
70  AllAxes data = AllAxes();
71  int16_t rawData[3];
72  Read(kDataRegister, sizeof(rawData), (uint8_t *)rawData);
73 
74  data.XAxis = rawData[0] * kGsPerLSB;
75  data.YAxis = rawData[1] * kGsPerLSB;
76  data.ZAxis = rawData[2] * kGsPerLSB;
77  return data;
78 }
