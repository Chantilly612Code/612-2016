14 /**
15  * Class for reading analog potentiometers. Analog potentiometers read
16  * in an analog voltage that corresponds to a position. The position is
17  * in whichever units you choose, by way of the scaling and offset
18  * constants passed to the constructor.
19  *
20  * @author Alex Henning
21  * @author Colby Skeggs (rail voltage)
22  */
23 class AnalogPotentiometer : public Potentiometer, public LiveWindowSendable {
24  public:
25  /**
26  * AnalogPotentiometer constructor.
27  *
28  * Use the fullRange and offset values so that the output produces
29  * meaningful values. I.E: you have a 270 degree potentiometer and
30  * you want the output to be degrees with the halfway point as 0
31  * degrees. The fullRange value is 270.0(degrees) and the offset is
32  * -135.0 since the halfway point after scaling is 135 degrees.
33  *
34  * This will calculate the result from the fullRange times the
35  * fraction of the supply voltage, plus the offset.
36  *
37  * @param channel The analog channel this potentiometer is plugged into.
38  * @param fullRange The scaling to multiply the voltage by to get a meaningful
39  * unit.
40  * @param offset The offset to add to the scaled value for controlling the
41  * zero value
42  */
43  explicit AnalogPotentiometer(int channel, double fullRange = 1.0,
44  double offset = 0.0);

47  double offset = 0.0);
