64 /**
65  * Creates a one-pole IIR low-pass filter of the form:
66  * y[n] = (1-gain)*x[n] + gain*y[n-1]
67  * where gain = e^(-dt / T), T is the time constant in seconds
68  *
69  * This filter is stable for time constants greater than zero
70  *
71  * @param source The PIDSource object that is used to get values
72  * @param timeConstant The discrete-time time constant in seconds
73  * @param period The period in seconds between samples taken by the user
74  */
75 LinearDigitalFilter LinearDigitalFilter::SinglePoleIIR(std::shared_ptr<PIDSource> source,
76  double timeConstant,
77  double period) {
78  double gain = std::exp(-period / timeConstant);
79  return LinearDigitalFilter(std::move(source), {1.0 - gain}, {-gain});
80 }

82 /**
83  * Creates a first-order high-pass filter of the form:
84  * y[n] = gain*x[n] + (-gain)*x[n-1] + gain*y[n-1]
85  * where gain = e^(-dt / T), T is the time constant in seconds
86  *
87  * This filter is stable for time constants greater than zero
88  *
89  * @param source The PIDSource object that is used to get values
90  * @param timeConstant The discrete-time time constant in seconds
91  * @param period The period in seconds between samples taken by the user
92  */
93 LinearDigitalFilter LinearDigitalFilter::HighPass(std::shared_ptr<PIDSource> source,
94  double timeConstant,
95  double period) {
96  double gain = std::exp(-period / timeConstant);
97  return LinearDigitalFilter(std::move(source), {gain, -gain}, {-gain});
98 }

100 /**
101  * Creates a K-tap FIR moving average filter of the form:
102  * y[n] = 1/k * (x[k] + x[k-1] + ... + x[0])
103  *
104  * This filter is always stable.
105  *
106  * @param source The PIDSource object that is used to get values
107  * @param taps The number of samples to average over. Higher = smoother but
108  * slower
109  */
110 LinearDigitalFilter LinearDigitalFilter::MovingAverage(std::shared_ptr<PIDSource> source,
111  unsigned int taps) {
112  assert(taps > 0);
113 
114  std::vector<double> gains(taps, 1.0 / taps);
115  return LinearDigitalFilter(std::move(source), gains, {});
116 }

121 double LinearDigitalFilter::Get() const {
122  double retVal = 0.0;
123 
124  // Calculate the new value
125  for (unsigned int i = 0; i < m_inputGains.size(); i++) {
126  retVal += m_inputs[i] * m_inputGains[i];
127  }
128  for (unsigned int i = 0; i < m_outputGains.size(); i++) {
129  retVal -= m_outputs[i] * m_outputGains[i];
130  }
131 
132  return retVal;
133 }

143 /**
144  * Calculates the next value of the filter
145  *
146  * @return The filtered value at this step
147  */
148 double LinearDigitalFilter::PIDGet() {
149  double retVal = 0.0;
150 
151  // Rotate the inputs
152  m_inputs.PushFront(PIDGetSource());
153 
154  // Calculate the new value
155  for (unsigned int i = 0; i < m_inputGains.size(); i++) {
156  retVal += m_inputs[i] * m_inputGains[i];
157  }
158  for (unsigned int i = 0; i < m_outputGains.size(); i++) {
159  retVal -= m_outputs[i] * m_outputGains[i];
160  }
161 
162  // Rotate the outputs
163  m_outputs.PushFront(retVal);
164 
165  return retVal;
166 }
