12 /**
13  * Note that the VictorSP uses the following bounds for PWM values. These values
14  * should work reasonably well for
15  * most controllers, but if users experience issues such as asymmetric behavior
16  * around
17  * the deadband or inability to saturate the controller in either direction,
18  * calibration is recommended.
19  * The calibration procedure can be found in the VictorSP User Manual available
20  * from Vex.
21  *
22  * 2.004ms = full "forward"
23  * 1.52ms = the "high end" of the deadband range
24  * 1.50ms = center of the deadband range (off)
25  * 1.48ms = the "low end" of the deadband range
26  * 0.997ms = full "reverse"
27  */
28 
29 /**
30  * Constructor for a VictorSP
31  * @param channel The PWM channel that the VictorSP is attached to. 0-9 are
32  * on-board, 10-19 are on the MXP port
33  */
34 VictorSP::VictorSP(uint32_t channel) : SafePWM(channel) {
35  SetBounds(2.004, 1.52, 1.50, 1.48, .997);
36  SetPeriodMultiplier(kPeriodMultiplier_1X);
37  SetRaw(m_centerPwm);
38  SetZeroLatch();
39 
40  HALReport(HALUsageReporting::kResourceType_VictorSP, GetChannel());
41  LiveWindow::GetInstance()->AddActuator("VictorSP", GetChannel(), this);
42 }
