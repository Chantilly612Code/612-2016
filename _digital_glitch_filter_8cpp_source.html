50 void DigitalGlitchFilter::DoAdd(DigitalSource *input, int requested_index) {
51  // Some sources from Counters and Encoders are null. By pushing the check
52  // here, we catch the issue more generally.
53  if (input) {
54  int32_t status = 0;
55  setFilterSelect(m_digital_ports[input->GetChannelForRouting()],
56  requested_index, &status);
57  wpi_setErrorWithContext(status, getHALErrorMessage(status));
58 
59  // Validate that we set it correctly.
60  int actual_index = getFilterSelect(
61  m_digital_ports[input->GetChannelForRouting()], &status);
62  wpi_assertEqual(actual_index, requested_index);
63 
64  HALReport(HALUsageReporting::kResourceType_DigitalInput,
65  input->GetChannelForRouting());
66  }
67 }
