187 /**
188  * Drive the motors at "outputMagnitude" and "curve".
189  * Both outputMagnitude and curve are -1.0 to +1.0 values, where 0.0 represents
190  * stopped and not turning. curve < 0 will turn left and curve > 0 will turn
191  * right.
192  *
193  * The algorithm for steering provides a constant turn radius for any normal
194  * speed range, both forward and backward. Increasing m_sensitivity causes
195  * sharper turns for fixed values of curve.
196  *
197  * This function will most likely be used in an autonomous routine.
198  *
199  * @param outputMagnitude The speed setting for the outside wheel in a turn,
200  * forward or backwards, +1 to -1.
201  * @param curve The rate of turn, constant for different forward speeds. Set
202  * curve < 0 for left turn or curve > 0 for right turn.
203  * Set curve = e^(-r/w) to get a turn radius r for wheelbase w of your robot.
204  * Conversely, turn radius r = -ln(curve)*w for a given value of curve and
205  * wheelbase w.
206  */
207 void RobotDrive::Drive(float outputMagnitude, float curve) {
208  float leftOutput, rightOutput;
209  static bool reported = false;
210  if (!reported) {
211  HALReport(HALUsageReporting::kResourceType_RobotDrive, GetNumMotors(),
212  HALUsageReporting::kRobotDrive_ArcadeRatioCurve);
213  reported = true;
214  }
215 
216  if (curve < 0) {
217  float value = log(-curve);
218  float ratio = (value - m_sensitivity) / (value + m_sensitivity);
219  if (ratio == 0) ratio = .0000000001;
220  leftOutput = outputMagnitude / ratio;
221  rightOutput = outputMagnitude;
222  } else if (curve > 0) {
223  float value = log(curve);
224  float ratio = (value - m_sensitivity) / (value + m_sensitivity);
225  if (ratio == 0) ratio = .0000000001;
226  leftOutput = outputMagnitude;
227  rightOutput = outputMagnitude / ratio;
228  } else {
229  leftOutput = outputMagnitude;
230  rightOutput = outputMagnitude;
231  }
232  SetLeftRightMotorOutputs(leftOutput, rightOutput);
233 }

402 /**
403  * Arcade drive implements single stick driving.
404  * This function lets you directly provide joystick values from any source.
405  * @param moveValue The value to use for fowards/backwards
406  * @param rotateValue The value to use for the rotate right/left
407  * @param squaredInputs If set, increases the sensitivity at low speeds
408  */
409 void RobotDrive::ArcadeDrive(float moveValue, float rotateValue,
410  bool squaredInputs) {
411  static bool reported = false;
412  if (!reported) {
413  HALReport(HALUsageReporting::kResourceType_RobotDrive, GetNumMotors(),
414  HALUsageReporting::kRobotDrive_ArcadeStandard);
415  reported = true;
416  }
417 
418  // local variables to hold the computed PWM values for the motors
419  float leftMotorOutput;
420  float rightMotorOutput;
421 
422  moveValue = Limit(moveValue);
423  rotateValue = Limit(rotateValue);
424 
425  if (squaredInputs) {
426  // square the inputs (while preserving the sign) to increase fine control
427  // while permitting full power
428  if (moveValue >= 0.0) {
429  moveValue = (moveValue * moveValue);
430  } else {
431  moveValue = -(moveValue * moveValue);
432  }
433  if (rotateValue >= 0.0) {
434  rotateValue = (rotateValue * rotateValue);
435  } else {
436  rotateValue = -(rotateValue * rotateValue);
437  }
438  }
439 
440  if (moveValue > 0.0) {
441  if (rotateValue > 0.0) {
442  leftMotorOutput = moveValue - rotateValue;
443  rightMotorOutput = std::max(moveValue, rotateValue);
444  } else {
445  leftMotorOutput = std::max(moveValue, -rotateValue);
446  rightMotorOutput = moveValue + rotateValue;
447  }
448  } else {
449  if (rotateValue > 0.0) {
450  leftMotorOutput = -std::max(-moveValue, rotateValue);
451  rightMotorOutput = moveValue + rotateValue;
452  } else {
453  leftMotorOutput = moveValue - rotateValue;
454  rightMotorOutput = -std::max(-moveValue, -rotateValue);
455  }
456  }
457  SetLeftRightMotorOutputs(leftMotorOutput, rightMotorOutput);
458 }

460 /**
461  * Drive method for Mecanum wheeled robots.
462  *
463  * A method for driving with Mecanum wheeled robots. There are 4 wheels
464  * on the robot, arranged so that the front and back wheels are toed in 45
465  * degrees.
466  * When looking at the wheels from the top, the roller axles should form an X
467  * across the robot.
468  *
469  * This is designed to be directly driven by joystick axes.
470  *
471  * @param x The speed that the robot should drive in the X direction.
472  * [-1.0..1.0]
473  * @param y The speed that the robot should drive in the Y direction.
474  * This input is inverted to match the forward == -1.0 that joysticks produce.
475  * [-1.0..1.0]
476  * @param rotation The rate of rotation for the robot that is completely
477  * independent of
478  * the translation. [-1.0..1.0]
479  * @param gyroAngle The current angle reading from the gyro. Use this to
480  * implement field-oriented controls.
481  */
482 void RobotDrive::MecanumDrive_Cartesian(float x, float y, float rotation,
483  float gyroAngle) {
484  static bool reported = false;
485  if (!reported) {
486  HALReport(HALUsageReporting::kResourceType_RobotDrive, GetNumMotors(),
487  HALUsageReporting::kRobotDrive_MecanumCartesian);
488  reported = true;
489  }
490 
491  double xIn = x;
492  double yIn = y;
493  // Negate y for the joystick.
494  yIn = -yIn;
495  // Compenstate for gyro angle.
496  RotateVector(xIn, yIn, gyroAngle);
497 
498  double wheelSpeeds[kMaxNumberOfMotors];
499  wheelSpeeds[kFrontLeftMotor] = xIn + yIn + rotation;
500  wheelSpeeds[kFrontRightMotor] = -xIn + yIn - rotation;
501  wheelSpeeds[kRearLeftMotor] = -xIn + yIn + rotation;
502  wheelSpeeds[kRearRightMotor] = xIn + yIn - rotation;
503 
504  Normalize(wheelSpeeds);
505 
506  m_frontLeftMotor->Set(wheelSpeeds[kFrontLeftMotor] * m_maxOutput,
507  m_syncGroup);
508  m_frontRightMotor->Set(wheelSpeeds[kFrontRightMotor] * m_maxOutput,
509  m_syncGroup);
510  m_rearLeftMotor->Set(wheelSpeeds[kRearLeftMotor] * m_maxOutput, m_syncGroup);
511  m_rearRightMotor->Set(wheelSpeeds[kRearRightMotor] * m_maxOutput,
512  m_syncGroup);
513 
514  if (m_syncGroup != 0) {
515  CANJaguar::UpdateSyncGroup(m_syncGroup);
516  }
517 
518  m_safetyHelper->Feed();
519 }

521 /**
522  * Drive method for Mecanum wheeled robots.
523  *
524  * A method for driving with Mecanum wheeled robots. There are 4 wheels
525  * on the robot, arranged so that the front and back wheels are toed in 45
526  * degrees.
527  * When looking at the wheels from the top, the roller axles should form an X
528  * across the robot.
529  *
530  * @param magnitude The speed that the robot should drive in a given direction.
531  * [-1.0..1.0]
532  * @param direction The direction the robot should drive in degrees. The
533  * direction and maginitute are
534  * independent of the rotation rate.
535  * @param rotation The rate of rotation for the robot that is completely
536  * independent of
537  * the magnitute or direction. [-1.0..1.0]
538  */
539 void RobotDrive::MecanumDrive_Polar(float magnitude, float direction,
540  float rotation) {
541  static bool reported = false;
542  if (!reported) {
543  HALReport(HALUsageReporting::kResourceType_RobotDrive, GetNumMotors(),
544  HALUsageReporting::kRobotDrive_MecanumPolar);
545  reported = true;
546  }
547 
548  // Normalized for full power along the Cartesian axes.
549  magnitude = Limit(magnitude) * sqrt(2.0);
550  // The rollers are at 45 degree angles.
551  double dirInRad = (direction + 45.0) * 3.14159 / 180.0;
552  double cosD = cos(dirInRad);
553  double sinD = sin(dirInRad);
554 
555  double wheelSpeeds[kMaxNumberOfMotors];
556  wheelSpeeds[kFrontLeftMotor] = sinD * magnitude + rotation;
557  wheelSpeeds[kFrontRightMotor] = cosD * magnitude - rotation;
558  wheelSpeeds[kRearLeftMotor] = cosD * magnitude + rotation;
559  wheelSpeeds[kRearRightMotor] = sinD * magnitude - rotation;
560 
561  Normalize(wheelSpeeds);
562 
563  m_frontLeftMotor->Set(wheelSpeeds[kFrontLeftMotor] * m_maxOutput,
564  m_syncGroup);
565  m_frontRightMotor->Set(wheelSpeeds[kFrontRightMotor] * m_maxOutput,
566  m_syncGroup);
567  m_rearLeftMotor->Set(wheelSpeeds[kRearLeftMotor] * m_maxOutput, m_syncGroup);
568  m_rearRightMotor->Set(wheelSpeeds[kRearRightMotor] * m_maxOutput,
569  m_syncGroup);
570 
571  if (m_syncGroup != 0) {
572  CANJaguar::UpdateSyncGroup(m_syncGroup);
573  }
574 
575  m_safetyHelper->Feed();
576 }
